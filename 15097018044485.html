定义-返回值格式

返回格式

{
    "isException":true|false
    "code":xxx
    "msg":xxx
    "data":
}

代码处理

<?php
/**
 * @author lishuo
 */

namespace App\Tools;


use Illuminate\Database\Eloquent\Collection;
use Illuminate\Database\Eloquent\Model;
use Illuminate\Pagination\LengthAwarePaginator;
use Illuminate\Support\Facades\Response;

/**
 * 返回格式化
 * Class ReturnFormat
 * @package App\Tools
 */
class ReturnFormat
{
    /**
     * @param array $data
     * @param string $state
     * @param string $message
     * @param $isException
     * @return mixed
     */
    public static function message($data = [], $state = 'SUCCESS', $message = '', $isException = false)
    {
        self::formatData($data);
        $data = !is_array($data) ? [] : $data;//返回值为true的时候直接返回空数组
        $code = self::getCodeByStates($state);
        $message = $message ? : self::getMessageByStates($state);
        return Response::json([
            'isException' => $isException,
            "code"        => $code,
            "msg"         => $message,
            "data"        => $data,
        ]);
    }

    /**
     * 数据格式化  处理分页，具体分页参考分页的标准定义
     * @param $data
     */
    private static function formatData(&$data)
    {
        if ($data instanceof LengthAwarePaginator) {
            $data = $data->toArray();
            $data['list'] = $data['data'];
            unset($data['data'], $data['next_page_url'], $data['path'], $data['prev_page_url'], $data['last_page'], $data['from'], $data['to']);
        }
    }

    private static function getCodeByStates($state)
    {
        return isset(config("code." . $state)['code']) ? config("code." . $state)['code'] : '500.';
    }

    private static function getMessageByStates($state)
    {
        return isset(config("code." . $state)['msg']) ? config("code." . $state)['msg'] : '未知错误';
    }
}
