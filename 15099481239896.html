定义-异常处理

修改laravel异常类handler.php
异常返回统一处理


<?php

namespace App\Exceptions;

use App\Tools\LogWrite\LogWrite;
use App\Tools\ReturnFormat;
use Exception;
use Illuminate\Auth\AuthenticationException;
use Illuminate\Foundation\Exceptions\Handler as ExceptionHandler;

class Handler extends ExceptionHandler
{
    /**
     * A list of the exception types that should not be reported.
     *
     * @var array
     */
    protected $dontReport = [
        \Illuminate\Auth\AuthenticationException::class,
        \Illuminate\Auth\Access\AuthorizationException::class,
        \Symfony\Component\HttpKernel\Exception\HttpException::class,
        \Illuminate\Database\Eloquent\ModelNotFoundException::class,
        \Illuminate\Session\TokenMismatchException::class,
        \Illuminate\Validation\ValidationException::class,
    ];

    /**
     * Report or log an exception.
     *
     * This is a great spot to send exceptions to Sentry, Bugsnag, etc.
     *
     * @param  \Exception  $exception
     * @return void
     */
    public function report(Exception $exception)
    {
        parent::report($exception);
    }

    /**
     * Render an exception into an HTTP response.
     *
     * @param  \Illuminate\Http\Request  $request
     * @param  \Exception  $exception
     * @return \Illuminate\Http\Response
     */
    public function render($request, Exception $exception)
    {
        $message = $this->parseException($exception);
        $errors = [
            "error_file" => $exception->getFile(),
            "error_code" => $exception->getCode(),
            'error_message' => $exception->getMessage(),
        ];
        LogWrite::error(array_merge($message, $errors), "异常错误",'error');
        return ReturnFormat::message([], $message['status'], $message['msg'], true);//统一返回参数不合
    }

    /**
     * Convert an authentication exception into an unauthenticated response.
     *
     * @param  \Illuminate\Http\Request  $request
     * @param  \Illuminate\Auth\AuthenticationException  $exception
     * @return \Illuminate\Http\Response
     */
    protected function unauthenticated($request, AuthenticationException $exception)
    {
        if ($request->expectsJson()) {
            return response()->json(['error' => 'Unauthenticated.'], 401);
        }

        return redirect()->guest('login');
    }
    /**
     * 解析失败
     * @return array
     */
    protected function parseException(Exception $exception)
    {
        $message = 'TNTERNAL_SERVER_ERROR';
        $msg = '';
        if ($exception instanceof LoginException || xxx 捕捉自动以异常
        ) {
            $message = $exception->getMessage();
        } elseif ($exception instanceof ConnectionException) {
            $message = 'REDIS_HANDLING_ERROR';
        } elseif ($exception instanceof NotFoundHttpException) {
            $message = 'NOT_FOUND';
        } elseif ($exception instanceof MethodNotAllowedHttpException) {
            $message = 'METHOD_NOT_ALLOWED';
        } elseif ($exception instanceof ValidationException) {
            $errors = $exception->validator->errors()->getMessages();
            foreach ($errors as $k => $v) {
                $msg = $k . '|' . $v[0];
                break;
            }
            $message = 'PARAM_INVALID';
        } elseif ($exception instanceof QueryException) {
            $message = 'DATA_HANDLING_ERROR';
        }
        LogWrite::exception($exception);
        return ['status' => $message, 'msg' =>$msg ];
    }
}
