table具体某一列值+bootstrap弹框

常用弹框实例

@extends('admin.layouts.base')

@section('afterHeadCss')
    <link rel="stylesheet" href="{{asset('/css/admin/productList.css')}}">
@append

@section('main')
    <div class="list-container">
        <div class="list-title">
            <p>平台账户中心</p>
        </div>
        <div class="list-container">
            <div class="table-container table-responsive">
                <table id="list-table" class="list-table table table-bordered table-hover">
                    <thead>
                    <tr>
                        <td>账户名称</td>
                        <td>账户类型</td>
                        <td>可用金额</td>
                        <td>冻结金额</td>
                        <td>银行编号</td>
                        <td>银行卡号</td>
                        <td>身份证号</td>
                        <td>预留手机号</td>
                        <td>认证状态</td>
                        <td>创建时间</td>
                        <td>操作</td>
                    </tr>
                    </thead>
                    <tbody class="table-content">
                    @foreach($data['list'] as $datum)
                        <tr>
                            <td>{{$datum['name']}}</td>
                            <td>{{config('enum.platFormWalletEnum.type.'.$datum['type'])}}</td>
                            <td>{{$datum['valid_cash']}}</td>
                            <td>{{$datum['freeze_cash']}}</td>
                            <td>{{$datum['bank_code']}}</td>
                            <td>{{$datum['card_number']}}</td>
                            <td>{{$datum['identity_number']}}</td>
                            <td>{{$datum['card_mobile']}}</td>
                            <td>{{config('enum.platFormWalletEnum.card_status.'.$datum['card_status'])}}</td>
                            <td>{{$datum['created_at']}}</td>
                            <td>
                                <input type="hidden" value="{{$datum['id']}}" name="id" class="id">
                                @if($datum['card_status']==0 && authVerify('admin-wallet/platformwallet-list'))
                                    <button
                                            class="btn btn-primary" data-toggle="modal" data-target="#myModal"
                                            data-toggle="modal" onclick="submitForm(this)">绑卡
                                    </button>
                                @endif
                            </td>
                        </tr>
                    @endforeach
                    </tbody>
                </table>
                <ul class="pager">
                    <li class="previous @if($data['current_page']<=1) disabled @endif"><a
                                href="/admin-wallet/platformwallet-list?page={{$data['older_page']}}">&larr; 上一页</a>
                    </li>
                    <li class="next @if($data['current_page']>=$data['total_page']) disabled @endif"><a
                                href="/admin-wallet/platformwallet-list?page={{$data['newer_page']}}">下一页 &rarr;</a>
                    </li>
                </ul>
            </div>
        </div>

        <!-- Modal -->
        <div class="modal fade" id="myModal" tabindex="-1" role="dialog" aria-labelledby="myModalLabel">
            <div class="modal-dialog" role="document">
                <div class="modal-content">
                    <div class="modal-header">
                        <button type="button" class="close" data-dismiss="modal" aria-label="Close"><span
                                    aria-hidden="true">&times;</span></button>
                        <h4 class="modal-title" id="myModalLabel">绑卡</h4>
                    </div>
                    <div class="modal-body">
                        <div class="row">
                            <form class="form-horizontal" method="post">
                                <div class="form-group">
                                    <input type="hidden" class="form-control" id="selectId">
                                    <label for="inputEmail3" class="col-sm-2 control-label">银行编号</label>
                                    <div class="col-sm-10">
                                        <input class="form-control" id="bank_code">
                                    </div>
                                </div>
                                <div class="form-group">
                                    <label for="inputPassword3" class="col-sm-2 control-label">银行账号</label>
                                    <div class="col-sm-10">
                                        <input class="form-control" id="card_number">
                                    </div>
                                </div>
                            </form>
                        </div>
                    </div>
                    <div class="modal-footer">
                        <button type="button" class="btn btn-default" data-dismiss="modal">关闭</button>
                        <input type="button" value="确定" class="btn btn-primary" onclick="submitBindCard();"/>
                    </div>
                </div>
            </div>
        </div>
        {{-- href="/admin-wallet/platformwallet-bindCard/{{$datum["id"]}}" type="button"--}}
    </div>

    <script>
        //弹框中的提交
        function submitBindCard() {
            var id = $('#selectId').val();
            var bank_code = $('#bank_code').val();
            var card_number = $('#card_number').val();
            $.post('/admin-wallet/platformwallet-bindCard/' + id,
                {
                    "bank_code": bank_code,
                    "card_number": card_number
                },
                function (data) {
                    if (data) {
                        console.log(data);
                    }
                });
        }
        //获取某个表单的值
        function submitForm(btn) {
            //获取参数表单参数
            var id = $(btn.parentElement.children[0]).val();
            $('#selectId').val(id);
            
            //还有一种根据名称去取值  $("input[name='recharge_id']").val();
            //同样所有的取值都可以复制 $("input[name='xxx']").val(123);
        }

    </script>
@stop
