public class User extends BasePojo {

    private Integer userId;

    @NotNull(message = "姓名不能为空")   //可以在请求的时候，如果这个参数不填会报错
    private String userName;

    //@NotNull(message = "默认的信息：不能为nul")  //会有好多中做法
    private String password;


//BasePojo  下面2个参数只是封装了分页的请求参数，如果后期有什么公共的参数都可以封装，这样后续
//参数都是以对象的形式，
public class BasePojo {
  
    @Transient   //如果是ORM框架操作数据库的时候不需要把这个字段做映射
    public int pageSize;
    @Transient
    public int pageNum;
    ...


//使用
//对于上面的错误信息
  public Result create(@Valid User user, BindingResult bindingResult) {
  //如果上面的验证没有通过的话。bindingResult  里面封装了错误信息
        if (bindingResult.hasErrors()) {
            String msg = bindingResult.getFieldError().getDefaultMessage();
            return ResultUtils.error(msg);
        }
        //返回创建的用户id
        return ResultUtils.success(userServices.create(user));
    }
